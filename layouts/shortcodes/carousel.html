"dark" -}} {{- $classes = printf "%s %s" $classes "carousel-dark" -}} {{- else if eq . "fade" -}} {{- $classes = printf "%s %s" $classes "carousel-fade" -}} {{- else if eq . "no_indicators" -}} {{- $indicators = false -}} {{- else if eq . "no_controls" -}} {{- $controls = false -}} {{- else if eq . "escape" -}} {{- $escape = true -}} {{ else }} {{ errorf "Unknown option to shortcode 'carousel': " . }} {{- end -}} {{- end -}}

{{/* Get the 'alt' description from the image as the heading of this slide */}} {{- $heading := replaceRE "" "$2" $img -}} {{- $headings = $headings | append $heading -}}
